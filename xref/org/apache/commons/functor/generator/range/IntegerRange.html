24  /**
25   * A range of integers.
26   *
27   * @since 1.0
28   * @version $Revision: 1385335 $ $Date: 2012-09-16 15:08:31 -0300 (Sun, 16 Sep 2012) $
29   */
30  public class IntegerRange extends NumericRange<Integer> {
31  
32      // attributes
33      // ---------------------------------------------------------------
34      /**
35       * Left limit.
36       */
37      private final Endpoint<Integer> leftEndpoint;
38  
39      /**
40       * Right limit.
41       */
42      private final Endpoint<Integer> rightEndpoint;
43  
44      /**
45       * Increment step.
46       */
47      private final int step;

49      /**
50       * Calculate default step.
51       */
52      public static final BinaryFunction<Integer, Integer, Integer> DEFAULT_STEP =
53              new BinaryFunction<Integer, Integer, Integer>() {
54  
55          public Integer evaluate(Integer left, Integer right) {
56              return left > right ? -1 : 1;
57          }
58      };

100     public IntegerRange(int from, int to, int step) {
101         this(from, DEFAULT_LEFT_BOUND_TYPE, to, DEFAULT_RIGHT_BOUND_TYPE, step);
102     }

104     /**
105      * Create a new IntegerRange.
106      *
107      * @param from start
108      * @param leftBoundType type of left bound
109      * @param to end
110      * @param rightBoundType type of right bound
111      * @param step increment
112      */
113     public IntegerRange(int from, BoundType leftBoundType, int to,
114                         BoundType rightBoundType, int step) {
115         this.leftEndpoint = Validate
116             .notNull(new Endpoint<Integer>(from, leftBoundType),
117                      "Left Endpoint argument must not be null");
118         this.rightEndpoint = Validate
119             .notNull(new Endpoint<Integer>(to, rightBoundType),
120                      "Right Endpoint argument must not be null");
121         this.step = step;
122         if (from != to && Integer.signum(step) != Integer.signum(to - from)) {
123             throw new IllegalArgumentException("Will never reach " + to
124                                                + " from " + from
125                                                + " using step " + step);
126         }
127     }
128 
129     /**
130      * Create a new IntegerRange.
131      *
132      * @param from start
133      * @param to end
134      * @param step increment
135      */
136     public IntegerRange(Endpoint<Integer> from, Endpoint<Integer> to, int step) {
137         this.leftEndpoint = Validate
138             .notNull(from, "Left Endpoint argument must not be null");
139         this.rightEndpoint = Validate
140             .notNull(to, "Right Endpoint argument must not be null");
141         this.step = step;
142         if (from != to
143             && Integer.signum(step) != Integer.signum(to.getValue()
144                                                    - from.getValue())) {
145             throw new IllegalArgumentException("Will never reach " + to
146                                                + " from " + from
147                                                + " using step " + step);
148         }
149     }

174     /**
175      * {@inheritDoc}
176      */
177     public void run(UnaryProcedure<? super Integer> proc) {
178         final int step = this.getStep();
179         final boolean includeLeftValue = this.getLeftEndpoint()
180             .getBoundType() == BoundType.CLOSED;
181         final boolean includeRightValue = this.getRightEndpoint()
182             .getBoundType() == BoundType.CLOSED;
183         final int leftValue = this.getLeftEndpoint().getValue();
184         final int rightValue = this.getRightEndpoint().getValue();
185         if (step < 0) {
186             final int from = includeLeftValue ? leftValue : leftValue + step;
187             if (includeRightValue) {
188                 for (int i = from; i >= rightValue; i += step) {
189                     proc.run(i);
190                 }
191             } else {
192                 for (int i = from; i > rightValue; i += step) {
193                     proc.run(i);
194                 }
195             }
196         } else {
197             final int from = includeLeftValue ? leftValue : leftValue + step;
198             if (includeRightValue) {
199                 for (int i = from; i <= rightValue; i += step) {
200                     proc.run(i);
201                 }
202             } else {
203                 for (int i = from; i < rightValue; i += step) {
204                     proc.run(i);
205                 }
206             }
207         }
208     }
