15  package org.apache.commons.functor.generator.util;
16  
17  import java.util.ArrayList;
18  import java.util.Collection;
19  
20  import org.apache.commons.functor.UnaryFunction;
21  import org.apache.commons.functor.UnaryProcedure;
22  import org.apache.commons.functor.generator.Generator;
23  import org.apache.commons.lang3.Validate;
24  
25  /**
26   * Transforms a generator into a collection. If a collection is not passed into
27   * the constructor an ArrayList will be returned from the transform method.
28   *
29   * @param <E> the type of elements held in the adapted collection.
30   * @param <C> the type of collection to be filled.
31   * @since 1.0
32   * @version $Revision: 1376354 $ $Date: 2012-08-23 00:04:55 -0300 (Thu, 23 Aug 2012) $
33   */
34  public class CollectionTransformer<E, C extends Collection<? super E>>
35                          implements UnaryFunction<Generator<? extends E>, C> {
36      /*
37       * TODO revisit this class... it could stand a more-descriptive name.  Also, it's a little
38       * hard to say whether, for an instance constructed without a specific target collection,
39       * #evaluate() should return a new ArrayList for each call, or continue adding to
40       * a single ArrayList instance (the current behavior).
41       * Perhaps this is more a documentation issue than anything.
42       */
43  
44      // instance methods
45      //---------------------------------------------------
46      /**
47       * The adapted collection has to be filled.
48       */
49      private final C toFill;
50  
51      // constructors
52      //---------------------------------------------------
53      /**
54       * Create a new CollectionTransformer.
55       * @param toFill Collection to fill
56       */
57      public CollectionTransformer(C toFill) {
58          this.toFill = Validate.notNull(toFill, "toFill");
59      }
60  
61      // instance methods
62      //---------------------------------------------------
63      /**
64       * {@inheritDoc}
65       */
66      public C evaluate(Generator<? extends E> generator) {
67          generator.run(new UnaryProcedure<E>() {
68              public void run(E obj) {
69                  toFill.add(obj);
70              }
71          });
72          return toFill;
73      }
74  
75      /**
76       * Get a {@link CollectionTransformer} instance that simply returns any {@link Collection}.
77       * @param <E> the returned collection type.
78       * @return {@link CollectionTransformer}
79       */
80      public static <E> CollectionTransformer<E, Collection<E>> toCollection() {
81          return new CollectionTransformer<E, Collection<E>>(new ArrayList<E>());
82      }
83  }
