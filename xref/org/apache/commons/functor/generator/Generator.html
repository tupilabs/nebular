14  package org.apache.commons.functor.generator;
15  
16  import java.util.Collection;
17  
18  import org.apache.commons.functor.UnaryFunction;
19  import org.apache.commons.functor.UnaryProcedure;
20  
21  /**
22   * The Generator interface defines a number of useful actions applying UnaryFunctors
23   * to each in a series of argument Objects.
24   *
25   * @param <E> the type of elements held in this generator.
26   * @version $Revision: 1376354 $ $Date: 2012-08-23 00:04:55 -0300 (Thu, 23 Aug 2012) $
27   */
28  public interface Generator<E> {
29      /**
30       * Generators must implement this method.
31       * @param proc UnaryProcedure to run
32       */
33      void run(UnaryProcedure<? super E> proc);
34  
35      /**
36       * Transforms this generator using the passed in
37       * transformer. An example transformer might turn the contents of the
38       * generator into a {@link Collection} of elements.
39       * @param <Z> the returned value type of the input {@link UnaryFunction}.
40       * @param transformer UnaryFunction to apply to this
41       * @return transformation result
42       */
43      <Z> Z to(UnaryFunction<Generator<? extends E>, ? extends Z> transformer);
44  
45      /**
46       * Same as to(new CollectionTransformer(collection)).
47       * @param <C> the returned collection type.
48       * @param collection Collection to which my elements should be added
49       * @return <code>collection</code>
50       */
51      <C extends Collection<? super E>> C to(C collection);
52  
53      /**
54       * Same as to(new CollectionTransformer()).
55       * @return Collection
56       */
57      Collection<? super E> toCollection();
58  }
