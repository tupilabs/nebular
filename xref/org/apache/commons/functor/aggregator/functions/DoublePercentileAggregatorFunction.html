17  package org.apache.commons.functor.aggregator.functions;
18  
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.apache.commons.functor.UnaryFunction;
24  
25  /**
26   * Aggregator function to be used with subclasses of
27   * {@link org.apache.commons.functor.aggregator.AbstractListBackedAggregator}
28   * which retrieves the <a
29   * href="http://en.wikipedia.org/wiki/Percentile">percentile</a> value for a
30   * given percentile. The percentile rank/index for a value P can be obtained
31   * using formula: <code>n = round((P / 100) * N + 0.5)</code> where N is the
32   * number of items in a list.
33   */
34  public class DoublePercentileAggregatorFunction implements UnaryFunction<List<Double>, Double> {
35      /** A percentile goes from 0 to 100% and that's it. */
36      private static final double MAX_PERCENTAGE = 100.0;
37      /**
38       * Percentile value to calculate. 0 &lt; percentile &lt;= 100
39       */
40      private double              percentile;
41  
42      /**
43       * Flag to indicate whether we are going to operate on a copy of the list
44       * given or not. In order to compute the percentile, we need to sort the
45       * list first (and then choose the item based on the formula given above).
46       * This function offers 2 ways of doing the sorting:
47       * <ul>
48       * <li>by sorting (modifying) the original list (<code>useCopy=false</code>)
49       * </li>
50       * <li>by operating on a copy of the original list and leaving the original
51       * untouched (<code>useCopy=true</code>)</li>
52       * </ul>
53       * NOTE: While using a copy ensures the original list is untouched, it does
54       * mean we are creating a temporary list for the purpose of this computation
55       * so it will have an impact on memory!
56       */
57      private boolean             useCopy;
58  
59      /**
60       * Similar to {@link #DoublePercentileAggregatorFunction(double, boolean)
61       * DoublePercentilAggregatorFunction(percentile,true)}.
62       *
63       * @param percentile
64       *            Percentile this function will return the value for
65       */
66      public DoublePercentileAggregatorFunction(double percentile) {
67          this(percentile, true);
68      }
69  
70      /**
71       * Initializes the function with the given percentile and decides whether
72       * the function will modify the original list or not.
73       *
74       * @param percentile
75       *            Percentile this function will return the value for
76       * @param useCopy
77       *            If set to true, the original list will not be modified and
78       *            will contain the data in sorted order, if false, this instance
79       *            will operate on a copy of the list
80       */
81      public DoublePercentileAggregatorFunction(double percentile, boolean useCopy) {
82          if (percentile < 0.0 || percentile > MAX_PERCENTAGE) {
83              throw new IllegalArgumentException("Invalid value for percentile: " + percentile);
84          }
85          this.percentile = percentile;
86          this.useCopy = useCopy;
87      }
88  
89      /**
90       * Used internally to compute the rank of the item in the list for the
91       * requested percentile. This is invoked internally from
92       * {@link #evaluate(List)}.
93       *
94       * @param data
95       *            List containing data. This cannot be <code>null</code> (throws
96       *            <code>NullPointerException</code>) or empty (throws
97       *            <code>ArrayIndexOutOfBoundsException</code>).
98       * @return Index of the item for the requested percentile
99       * @see #getPercentile()
100      */
101     final int computeRank(List<Double> data) {
102         int maxRank = data.size() - 1;
103         int rank = (int) Math.floor((percentile * maxRank) / MAX_PERCENTAGE);
104         return rank;
105     }
106 
107     /**
108      * Traverses the list and computes the percentile. In doing so, it sorts the
109      * list first -- and might or might not use the original list or a copy
110      * depending on {@link #isUseCopy()}.
111      *
112      * @param data
113      *            List to compute the percentile for
114      * @return percentile of the given list or null if list is <code>null</code>
115      *         or empty (zero size).
116      */
117     public Double evaluate(List<Double> data) {
118         if (data == null || data.size() == 0) {
119             return null;
120         }
121         List<Double> copy = data;
122         if (useCopy) {
123             copy = new ArrayList<Double>(data);
124         }
125         Collections.sort(copy);
126         int rank = computeRank(data);
127         return copy.get(rank);
128     }
129 
130     /**
131      * Getter for {@link #percentile}.
132      *
133      * @return Retrieves the percentile this instance will return
134      */
135     public double getPercentile() {
136         return percentile;
137     }
138 
139     /**
140      * Does this instance modify the passed in list or not?
141      *
142      * @return If true, this instance will sort the list passed in and then use
143      *         it to compute the percentile; if false, it will operate on a copy
144      *         of the list
145      * @see DoublePercentileAggregatorFunction#DoublePercentileAggregatorFunction(double,
146      *      boolean)
147      */
148     public boolean isUseCopy() {
149         return useCopy;
150     }
151 
152     @Override
153     public String toString() {
154         return DoublePercentileAggregatorFunction.class.getName();
155     }
156 }
