17  package org.apache.commons.functor.aggregator.functions;
18  
19  import org.apache.commons.functor.BinaryFunction;
20  
21  /**
22   * Aggregator function to be used with subclasses of
23   * {@link org.apache.commons.functor.aggregator.AbstractNoStoreAggregator} which
24   * simply increments the first argument by 1 and returns it. The reason behind
25   * this class is to provide a simple counter implementation: users call
26   * {@link org.apache.commons.functor.aggregator.AbstractNoStoreAggregator#add(Object)}
27   * with some data (which will be ignored) and the class will simply count how
28   * many times this call was made. Such an implementation can be used for
29   * instance to keep track for instance of number of transactions going through a
30   * system, whereas every time a transaction was made we just increment a
31   * counter. This can be achieved still using a
32   * {@link org.apache.commons.functor.aggregator.AbstractNoStoreAggregator} and a
33   * {@link DoubleSumAggregatorBinaryFunction}-like function (but using int) and always
34   * supplying the second parameter as 1 (one). However, using this might make the
35   * code clearer.
36   */
37  public final class IntegerCountAggregatorBinaryFunction implements BinaryFunction<Integer, Integer, Integer> {
38      /**
39       * Increments <code>left</code> by one and returns it.
40       *
41       * @param left
42       *            Value to be incremented by 1 and returned. If
43       *            <code>null</code>, <code>null</code> is returned.
44       * @param right
45       *            ignored
46       * @return <code>left + 1</code> if <code>left != null</code> otherwise it
47       *         returns <code>null</code>.
48       */
49      public Integer evaluate(Integer left, Integer right) {
50          if (left == null) {
51              return null;
52          }
53          return left + 1;
54      }
55  
56      @Override
57      public String toString() {
58          return IntegerCountAggregatorBinaryFunction.class.getName();
59      }
60  }
