17  package org.apache.commons.functor.aggregator;
18  
19  import java.util.List;
20  
21  import org.apache.commons.functor.UnaryFunction;
22  import org.apache.commons.lang3.Validate;
23  
24  /**
25   * An aggregator which stores the data series in a List. Every call to
26   * {@link #add(Object)} will add an item in the array and there is no limit to
27   * how much data we can store. It is down to subclasses to decide which types of
28   * List implementation they need to used -- and the abstract factory
29   * {@link #createList()} is provided for this.
30   * <p>This implementation also allows for various "aggregations" of the list to be
31   * used by providing a {@link UnaryFunction UnaryFunction<List<T>, T>} in the
32   * constructor.</p>
33   * <p>
34   * <b>Thread safety</b> : Note that due to the fact that
35   * {@link AbstractTimedAggregator} provides a threadsafe environment for access
36   * to data, the <code>List</code> implementation can be unsynchronized.
37   * </p>
38   *
39   * @param <T>
40   *            Type of object stored.
41   * @see AbstractTimedAggregator
42   */
43  public abstract class AbstractListBackedAggregator<T> extends AbstractTimedAggregator<T> {
44      /**
45       * Stores the data series we ought to aggregate/evaluate. This list can only
46       * be modified via {@link #reset()} and {@link #add(Object)} and will be
47       * traversed during {@link #evaluate()}.
48       */
49      private List<T>                   series;
50  
51      /**
52       * Used to actually aggregate the data when {@link #evaluate()} is called.
53       * This is set in {@link #AbstractListBackedAggregator() the constructor}.
54       */
55      private UnaryFunction<List<T>, T> aggregationFunction;
56  
57      /**
58       * Default constructor. Similar to
59       * {@link #AbstractListBackedAggregator(UnaryFunction, long)
60       * AbstractListBackedAggregator(aggregationFunction,0L}.
61       *
62       * @param aggregationFunction
63       *            Aggregation function to use in {@link #evaluate()}. Throws
64       *            <code>NullPointerException</code> if this is <code>null</code>
65       */
66      public AbstractListBackedAggregator(UnaryFunction<List<T>, T> aggregationFunction) {
67          this(aggregationFunction, 0L);
68      }
69  
70      /**
71       * Similar to
72       * {@link #AbstractListBackedAggregator(UnaryFunction, long, boolean)
73       * AbstractListBackedAggregator(aggregationFunction,interval,false}.
74       *
75       * @param aggregationFunction
76       *            Aggregation function to use in {@link #evaluate()}. Throws
77       *            <code>NullPointerException</code> if this is <code>null</code>
78       * @param interval
79       *            interval in miliseconds to reset this aggregator
80       */
81      public AbstractListBackedAggregator(UnaryFunction<List<T>, T> aggregationFunction, long interval) {
82          this(aggregationFunction, interval, false);
83      }
84  
85      /**
86       * Constructs an aggregator which will use the given function, reset itself
87       * at the given interval and will use a shared timer on own private timer.
88       *
89       * @param aggregationFunction
90       *            Aggregation function to use in {@link #evaluate()}. Throws
91       *            <code>NullPointerException</code> if this is <code>null</code>
92       * @param interval
93       *            interval in miliseconds to reset this aggregator
94       * @param useSharedTimer
95       *            if set to true, it will use a shared timer, as per
96       *            {@link AbstractTimedAggregator#AbstractTimedAggregator(long, boolean)}
97       *            ; otherwise if it's false it will use its own timer instance
98       * @see AbstractTimedAggregator#AbstractTimedAggregator(long, boolean)
99       */
100     public AbstractListBackedAggregator(UnaryFunction<List<T>, T> aggregationFunction, long interval,
101             boolean useSharedTimer) {
102         super(interval, useSharedTimer);
103         this.aggregationFunction = Validate.notNull(aggregationFunction, "UnaryFunction argument must not be null");
104         this.series = createList();
105     }
106 
107     /**
108      * Adds data to the series which will be aggregated. This implementation
109      * simply adds the data to the {@link #series} list.
110      *
111      * @param data
112      *            Data to be added to the data series.
113      */
114     @Override
115     public final void doAdd(T data) {
116         series.add(data);
117     }
118 
119     /**
120      * The actual "beef" of this class: iterate through the list and aggregates
121      * all the data and evaluates the result. This is done by calling
122      * <code>aggregationFunction.evaluate(series)</code>.
123      *
124      * @return the result of <code>aggregationFunction.evaluate(series)</code>
125      * @see Aggregator#evaluate()
126      */
127     @Override
128     protected final T doEvaluate() {
129         return aggregationFunction.evaluate(series);
130     }
131 
132     /**
133      * Resets the data series to the empty state.
134      */
135     @Override
136     protected final void doReset() {
137         series.clear();
138     }
139 
140     /**
141      * Allows subclasses to create the list which will store the {@link #series
142      * data series}.
143      *
144      * @return an instance of <code>List</code> which will be used to store the
145      *         data.
146      */
147     protected abstract List<T> createList();
148 
149     /**
150      * Getter for {@link #series}.
151      *
152      * @return Value of {@link #series}
153      */
154     protected final List<T> getSeries() {
155         return series;
156     }
157 
158     /**
159      * Simply returns the size of the data series which is the size of the list
160      * used internally.
161      *
162      * @return Size of {@link #series} -- equivalent to
163      *         <code>series.size()</code>
164      */
165     @Override
166     protected final int retrieveDataSize() {
167         return series.size();
168     }
169 
170     /**
171      * Getter for {@link #aggregationFunction}. Provided for testing purposes
172      * only.
173      *
174      * @return Current value of {@link #aggregationFunction}
175      */
176     final UnaryFunction<List<T>, T> getAggregationFunction() {
177         return aggregationFunction;
178     }
179 
180     @Override
181     public String toString() {
182         return AbstractListBackedAggregator.class.getName();
183     }
184 }
