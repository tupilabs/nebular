17  package org.apache.commons.functor.core.collection;
18  
19  import java.util.Iterator;
20  
21  import org.apache.commons.functor.UnaryFunction;
22  import org.apache.commons.lang3.Validate;
23  
24  /**
25   * Iterator that transforms another Iterator by applying a UnaryFunction to each returned element.
26   *
27   * @param <E> the function argument type
28   * @param <T> the iterator elements type
29   * @version $Revision: 1345136 $ $Date: 2012-06-01 09:47:06 -0300 (Fri, 01 Jun 2012) $
30   */
31  public final class TransformedIterator<E, T> implements Iterator<T> {
32  
33      // attributes
34      // ------------------------------------------------------------------------
35  
36      /**
37       * The function to apply to each iterator element.
38       */
39      private final UnaryFunction<? super E, ? extends T> function;
40      /**
41       * The wrapped iterator.
42       */
43      private final Iterator<? extends E> iterator;
44  
45      // constructor
46      // ------------------------------------------------------------------------
47      /**
48       * Create a new TransformedIterator.
49       * @param iterator Iterator to decorate
50       * @param function to apply
51       */
52      public TransformedIterator(Iterator<? extends E> iterator, UnaryFunction<? super E, ? extends T> function) {
53          this.function = Validate.notNull(function, "filtering UnaryFunction argument was null");
54          this.iterator = Validate.notNull(iterator, "Iterator argument was null");
55      }
56  
57      // iterator methods
58      // ------------------------------------------------------------------------
59  
60      /**
61       * {@inheritDoc}
62       * @see java.util.Iterator#hasNext()
63       */
64      public boolean hasNext() {
65          return iterator.hasNext();
66      }
67  
68      /**
69       * {@inheritDoc}
70       * @see java.util.Iterator#next()
71       */
72      public T next() {
73          return function.evaluate(iterator.next());
74      }
75  
76      /**
77       * {@inheritDoc}
78       * @see java.util.Iterator#remove()
79       */
80      public void remove() {
81          iterator.remove();
82      }
83  
84      /**
85       * {@inheritDoc}
86       */
87      @Override
88      public boolean equals(Object obj) {
89          if (obj == this) {
90              return true;
91          }
92          if (!(obj instanceof TransformedIterator<?, ?>)) {
93              return false;
94          }
95          TransformedIterator<?, ?> that = (TransformedIterator<?, ?>) obj;
96          return function.equals(that.function) && iterator.equals(that.iterator);
97      }
98  
99      /**
100      * {@inheritDoc}
101      */
102     @Override
103     public int hashCode() {
104         int hash = "TransformedIterator".hashCode();
105         hash <<= 2;
106         hash ^= function.hashCode();
107         hash <<= 2;
108         hash ^= iterator.hashCode();
109         return hash;
110     }
111 
112     /**
113      * {@inheritDoc}
114      */
115     @Override
116     public String toString() {
117         return "TransformedIterator<" + iterator + "," + function + ">";
118     }
119 
120     // class methods
121     // ------------------------------------------------------------------------
122     /**
123      * Get a Transformed Iterator instance.
124      *
125      * @param <E> the function argument type
126      * @param <T> the iterator elements type
127      * @param iter to decorate, if null result is null
128      * @param func transforming function, cannot be null
129      * @return Iterator<T>
130      */
131     public static <E, T> Iterator<T> transform(Iterator<? extends E> iter, UnaryFunction<? super E, ? extends T> func) {
132         if (null == iter) {
133             return null;
134         }
135         return new TransformedIterator<E, T>(iter, func);
136     }
137 
138     /**
139      * Get an Iterator instance that may be transformed.
140      *
141      * @param <E> the iterator elements type
142      * @param iter to decorate, if null result is null
143      * @param func transforming function, if null result is iter
144      * @return Iterator<?>
145      */
146     public static <E> Iterator<?> maybeTransform(Iterator<? extends E> iter, UnaryFunction<? super E, ?> func) {
147         return null == func ? (null == iter ? null : iter) : new TransformedIterator<E, Object>(iter, func);
148     }
