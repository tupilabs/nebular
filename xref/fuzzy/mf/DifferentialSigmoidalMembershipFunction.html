14  package fuzzy.mf;
15  
16  import org.apache.commons.math3.analysis.function.Sigmoid;
17  
18  /**
19   * Differential Sigmoidal Membership Function. Equivalent to Matlab
20   * <a href="http://www.mathworks.com/help/toolbox/fuzzy/dsigmf.html">dsigmf</a>
21   * function.
22   *
23   * @since 0.1
24   */
25  public class DifferentialSigmoidalMembershipFunction implements MembershipFunction<Double> {
26  
27  	protected final static double DEFAULT_LOW_ASYMPTOTE = 0.0;
28  	protected final static double DEFAULT_HIGH_ASYMPTOTE = 1.0;
29  
30  	protected transient Sigmoid sigmoid;
31  
32  	private final double a1;
33  	private final double c1;
34  	private final double a2;
35  	private final double c2;
36  
37  	public DifferentialSigmoidalMembershipFunction(double a1, double c1, double a2, double c2) {
38  		this(DEFAULT_LOW_ASYMPTOTE, DEFAULT_HIGH_ASYMPTOTE, a1, c1, a2, c2);
39  	}
40  
41  	public DifferentialSigmoidalMembershipFunction(double lowAsymptote, double highAsymptote, double a1, double c1, double a2, double c2) {
42  		sigmoid = new Sigmoid(lowAsymptote, highAsymptote);
43  		this.a1 = a1;
44  		this.c1 = c1;
45  		this.a2 = a2;
46  		this.c2 = c2;
47  	}
48  
49  	/*
50  	 * (non-Javadoc)
51  	 * @see fuzzy.mf.MembershipFunction#apply(java.lang.Object)
52  	 */
53  	@Override
54  	public Double apply(Double x) {
55  		final double r1 = sigmoid.value(a1*(x-c1));
56  		final double r2 = sigmoid.value(a2*(x-c2));
57  
58  		return r1 - r2;
59  	}
60  
61  	/* (non-Javadoc)
62  	 * @see java.lang.Object#equals(java.lang.Object)
63  	 */
64  	@Override
65  	public boolean equals(Object obj) {
66  		if(obj == null) {
67  			return false;
68  		}
69  		if(obj == this) {
70  			return true;
71  		}
72  		if(!(obj instanceof DifferentialSigmoidalMembershipFunction)) {
73  			return false;
74  		}
75  		final DifferentialSigmoidalMembershipFunction that = (DifferentialSigmoidalMembershipFunction)obj;
76  		return this.a1 == that.a1 && this.c1 == that.c1 && this.a2 == that.a2 && this.c2 == that.c2;
77  	}
78  
79  	/* (non-Javadoc)
80  	 * @see java.lang.Object#hashCode()
81  	 */
82  	@Override
83  	public int hashCode() {
84  		int hash = "DifferentialSigmoidalMembershipFunction".hashCode();
85  		hash <<= 2;
86  		hash ^= (int)this.a1;
87  		hash <<= 2;
88  		hash ^= (int)this.c1;
89  		hash <<= 2;
90  		hash ^= (int)this.a2;
91  		hash <<= 2;
92  		hash ^= (int)this.c2;
93  		return hash;
94  	}
95  
96  	/* (non-Javadoc)
97  	 * @see java.lang.Object#toString()
98  	 */
99  	@Override
100 	public String toString() {
101 		return "Differential Sigmoidal Membership Function ["+a1+" "+c1+" "+a2+" "+c2+"]";
102 	}
