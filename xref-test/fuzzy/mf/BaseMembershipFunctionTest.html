14  package fuzzy.mf;

16  import static org.junit.Assert.assertEquals;
17  import static org.junit.Assert.assertFalse;
18  import static org.junit.Assert.assertNotNull;
19  import static org.junit.Assert.assertTrue;

21  import java.util.function.Function;
22  
23  import org.junit.Test;
24  
25  /**
26   * Base test for Membership Functions.
27   * 
28   * @param <T> Type of Membership Function
29   * @since 0.1
30   */
31  public abstract class BaseMembershipFunctionTest<T extends Function<?, ?>> {
32  
33  	/**
34  	 * Creates a membership function for tests.
35  	 * @return membership function
36  	 */
37  	protected abstract T makeMembershipFunction();
38  	
39  	@Test
40  	public void testEqualsObject() {
41  		Object obj = makeMembershipFunction();
42  		assertEquals("Object must be reflexive", obj, obj);
43  		assertEquals("hashCode must be reflexive", obj.hashCode(), obj.hashCode());
44  		assertFalse(obj.equals(null));
45  		
46  		Object other = makeMembershipFunction();
47  		if(obj.equals(other)) {
48  			assertEquals("equals implies hash equals", obj.hashCode(), other.hashCode());
49  			assertEquals("equals must be symmetric", other, obj);
50  		} else {
51  			assertFalse("equals must be symmetric", other.equals(obj));
52  		}
53  		
54  		assertFalse(obj.equals(Boolean.FALSE));
55  	}
56  	
57  	@Test
58      public void testToStringIsOverridden() throws Exception {
59          Object obj = makeMembershipFunction();
60          assertNotNull("toString should never return null",obj.toString());
61          assertTrue(
62              obj.getClass().getName()  + " should override toString(), found \"" + obj.toString() + "\"",
63              !obj.toString().equals(objectToString(obj)));
64      }
65  	
66  	// private utils
67      // ------------------------------------------------------------------------
68      private String objectToString(Object obj) {
69          return obj.getClass().getName() + "@" + Integer.toHexString(obj.hashCode());
70      }
